    public static void bubbleSort(String[] vetor){
        for(int i = 0; i < (vetor.length - 1); i++){
            for (int j = 0; j < vetor.length - i - 1; j++){
                if(vetor[j].compareTo(vetor[j+1]) > 0){
                    String aux = vetor[j];
                    vetor[j] = vetor[j+1];
                    vetor[j+1] = aux;
                }
            }
        }
    }
